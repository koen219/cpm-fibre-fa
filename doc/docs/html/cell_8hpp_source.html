342  inline double AddSiteToMoments(int x, int y, double new_l = -1.) {
343 
344  // Add a site to the raw moments, then update and return the
345  // length of the cell
346 
347  // sum_x, sum_y, sum_xx, sum_xy and sum_yy are adjusted
348  // Eventually this function may be used to carry
349  // out all necessary adminstration at once

352  sum_xx += x * x;
353  sum_yy += y * y;
354  sum_xy += x * y;

356  // update length (see appendix. A, Zajac.jtb03), if length is not given

371  // Remove a site from the raw moments, then update and return the

394  inline double Length(long int s_x, long int s_y, long int s_xx, long int s_yy,
395  long int s_xy, long int n) {
396  // prevent NaN when last pixel is deleted
397  if (n == 0) {
398  return 0.;
399  }
400 
401  // prevent NaN when last pixel is deleted
402  if (n == 0) {
403  return 0.;
404  }
405 
406  // inertia tensor (constructed from the raw momenta, see notebook)
407  double iyy = (double)s_xx - (double)s_x * s_x / (double)n;
408  double ixx = (double)s_yy - (double)s_y * s_y / (double)n;
409  double ixy = -(double)s_xy + (double)s_x * s_y / (double)n;
410 
411  double rhs1 = (ixx + iyy) / 2.,
412  rhs2 = sqrt((ixx - iyy) * (ixx - iyy) + 4 * ixy * ixy) / 2.;
413 
414  double lambda_b = rhs1 + rhs2;
415  // double lambda_a=rhs1-rhs2;

431  inline void MajorMinorAxis(double *major_axis, double *minor_axis, double *v1,
432  double *v2) {
433 
434  // inertia tensor (constructed from the raw momenta, see notebook)
435  double iyy = (double)sum_xx - (double)sum_x * sum_x / (double)area;
436  double ixx = (double)sum_yy - (double)sum_y * sum_y / (double)area;
437  double ixy = -(double)sum_xy + (double)sum_x * sum_y / (double)area;
438 
439  double rhs1 = (ixx + iyy) / 2.,
440  rhs2 = sqrt((ixx - iyy) * (ixx - iyy) + 4 * ixy * ixy) / 2.;
441 
442  double lambda_b = rhs1 + rhs2;
443  double lambda_a = rhs1 - rhs2;
444 
445  // see: http://scienceworld.wolfram.com/physics/MomentofInertiaEllipse.html
446  // std::cerr << "n = " << n << "\n";
447  *major_axis = 4 * sqrt(lambda_b / area);
448  *minor_axis = 4 * sqrt(lambda_a / area);
449 
450  if (ixy != 0) {
451  *v1 = lambda_a - iyy;
452  *v2 = ixy;
453  } else {
454  *v1 = 0;
455  *v2 = 1;
456  }
457 
458  // normalize vector
459  double norm = sqrt((*v1) * (*v1) + (*v2) * (*v2));
460  *v1 /= norm;
461  *v2 /= norm;
462 
463  // 2*sqrt(lambda_b/n) give semimajor axis. We want the length.
464  }
